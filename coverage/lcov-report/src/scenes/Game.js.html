  shuffleDeck(deck) {
    const secureRandom = () => {
      const array = new Uint32Array(1);
      crypto.getRandomValues(array);
      return array[0] / 4294967296;
    };
    for (let i = deck.length - 1; i > 0; i--) {
      const j = Math.floor(secureRandom() * (i + 1));
      [deck[i], deck[j]] = [deck[j], deck[i]];
    }
    return deck;
  }
